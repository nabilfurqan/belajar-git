-- untuk melihat branch saat ini
git branch --show-current

-- ingin membuat branch baru
git branch nama-branch-baru
cth : git branch dev

-- ingin melihat semua branch
git branch --list atau git branch

-- ingin mengganti posisi branch
git switch nama-branch
cth : git switch dev

-- ingin rename nama branch
git branch -m branch-baru

-- menghapus branch
git branch -d nama-branch / -d = -delete

-- merge branch
git merge nama-branch

-- merge conflict / membatalkan merge
git merge --abort
